#include<stdio.h>
#include<stdlib.h>
void insersion(int *arr,int *n,int pos,int data)
{
    for(int i = *n;i >= pos  ; i--)
    {
        *(arr + i ) = *(arr + i - 1 );
    }
    (*n)++;
    *(arr + pos) = data;
     
}
void delete(int *arr,int *n, int pos)
{
    for(int i=pos;i<*n;i++)
    {
        *(arr + i) = *(arr + i +1);
    }
    (*n)--;
}
void traverse(int *arr,int *n)
{
    for(int i=0;i<*n;i++)
    {
        printf("%d\n",*(arr + i));
    }
}
void search(int *arr,int n,int key)
{
    for(int i=0;i<n;i++)
    {
        if(*(arr + i) == key)
        {
            printf("%d",i);
            break;
        }
    }
    
        
}

    int size;

    scanf("%d",&size);
    int arr[size];
    printf("Enter The Elements Of Array\n");
    for (int i = 0; i < size; i++)
    {
        scanf("%d",&arr[i]);
    }
    int pos;
    int data;
    int key;
    
    switch (operation)
    {
    case 1:
        
        printf("Enter the value of pos and data\n");
        scanf("%d %d",&pos,&data);
        insersion(arr,&size,pos,data);
        traverse(arr,&size);
        break;
    case 2:
        printf("Enter The pos\n");
        scanf("%d",&pos);

        traverse(arr,&size);

}
